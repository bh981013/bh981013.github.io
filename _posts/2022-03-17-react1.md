# ReactDOM.render

```js
function tick(){
  const elem = (
     <div>
      <h1>my first react</h1>
      <h2>time {new Date().toLocaleTimeString()}.</h2>
     </div>
  );
  ReactDOM.render(
    elem, document.getElementById('root')
  );
}

setInterval(tick, 1000);
```
